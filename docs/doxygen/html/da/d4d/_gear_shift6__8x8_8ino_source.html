341 void checkHistory() {
342  if (previousGears.isFull()) {
343  char gearHistory[BUFFER_SIZE]; // create new char array from history for comparison
344  for (int8_t i = 0; i < BUFFER_SIZE; i++) { // loop to populate array with char equivalents
345  gearHistory[i] = GearChars[previousGears[i]];
346  }
347  if (checkArrays(gearHistory, ANIM_SEQUENCE, BUFFER_SIZE) == true) { // compares the two arrays; if buffer history matches ANIM_SEQUENCE, then display animation
348  displayAnimation(random(ARRAY_SIZE(sprite) - 1)); // selects and displays random animation from struct array
349  }
350  else if (checkArrays(gearHistory, SCROLLTEXT_SEQUENCE, BUFFER_SIZE) == true) {
351  Parola.displayClear();
352  Parola.displayScroll(StartupText, PA_LEFT, PA_SCROLL_LEFT, SCROLL_SPEED); // scroll StartupText
353  while (!Parola.displayAnimate()) // play animation once until complete
354  ;
